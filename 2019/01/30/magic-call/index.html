FF的noSuchMethod没普及开，要实现magic_call的效果可以使用Proxy达到目的。

class MagicClass {
constructor() {
return new Proxy(this, {
get(target, name) {
if (name in target) {
return target[name]
}
return (…args) => {
args.unshift(name)
target.noSuchMethod.apply(target, args)
}
}
});
}
noSuchMethod(name, …args) {
console.log(name, args)
}
}
let inst = new MagicClass
inst.hello(‘a’)
